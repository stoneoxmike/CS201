	f(n) = 2^n 								//write formula for n
	f(n+1) = 2^(n+1) = (2^n)(2^1) = 2(2^n) 	//write formula for n+1
	f(n+1) = 2 * f(n) 						//write recursion for f(n+1) with f(n)
	f(n+1) = 2^n * 2 = 2^(n+1) 				//substitute f(n) into recursion and do algebra
